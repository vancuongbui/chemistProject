    function searchMode(term) {

        var mode ;
        var singleCharacter;
       
        term.split('%20').forEach(function (value, index) {
           
            if (value.length==1 || value.length==2) {
                               
                singleCharacter = true;
            }           
        })
        
        if (singleCharacter==true ) {

            mode = "allwords";           
        }
        else {
            mode= "anywords";
        }
        return mode;
